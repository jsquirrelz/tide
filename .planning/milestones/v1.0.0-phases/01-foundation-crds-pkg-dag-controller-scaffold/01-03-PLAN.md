Create the analyzer at `tools/analyzers/crosspool/analyzer.go`:
```go
// Package crosspool implements an analyzer that rejects any select statement
// that waits on both the planner and executor pool channels in the same case set.
//
// POOL-03 / Pitfall 6 prevention: the spec mandates two separately-sized
// parallelism semaphores (plannerPool, executorPool) that must not be silently
// unified. A select that waits on both is the canonical "I'm about to unify
// them" smell.
package crosspool

import (
    "go/ast"
    "strings"

    "golang.org/x/tools/go/analysis"
)

var Analyzer = &analysis.Analyzer{
    Name: "crosspool",
    Doc:  "rejects select statements that wait on both planner and executor pools (POOL-03 / Pitfall 6 prevention)",
    Run:  run,
}

func run(pass *analysis.Pass) (interface{}, error) {
    for _, f := range pass.Files {
        ast.Inspect(f, func(n ast.Node) bool {
            sel, ok := n.(*ast.SelectStmt)
            if !ok {
                return true
            }
            hasPlanner := false
            hasExecutor := false
            for _, commStmt := range sel.Body.List {
                cc, ok := commStmt.(*ast.CommClause)
                if !ok {
                    continue
                }
                if matchPoolIdent(cc.Comm, "planner") {
                    hasPlanner = true
                }
                if matchPoolIdent(cc.Comm, "executor") {
                    hasExecutor = true
                }
            }
            if hasPlanner && hasExecutor {
                pass.Reportf(sel.Pos(),
                    "cross-pool wait: select waits on both planner and executor pools (POOL-03 / Pitfall 6 violation)")
            }
            return true
        })
    }
    return nil, nil
}

// matchPoolIdent checks if a comm clause's statement references an identifier
// whose name contains `want` (case-insensitive). Handles both SendStmt (channel
// send) and ExprStmt-wrapped UnaryExpr (channel receive).
func matchPoolIdent(stmt ast.Stmt, want string) bool {
    if stmt == nil {
        return false
    }
    found := false
    ast.Inspect(stmt, func(n ast.Node) bool {
        if found {
            return false
        }
        ident, ok := n.(*ast.Ident)
        if !ok {
            return true
        }
        if strings.Contains(strings.ToLower(ident.Name), want) {
            found = true
            return false
        }
        return true
    })
    return found
}
```

Note on the `// want` comment: `analysistest.Run` matches Go-regex against the diagnostic message. The pattern is enclosed in backticks. The match is "diagnostic contains substring matching regex." Using the literal substring `cross-pool wait` is sufficient.

The position of the `// want` comment matters: it must be on the line where `pass.Reportf(sel.Pos(), ...)` reports — that's the `select` keyword's line.
