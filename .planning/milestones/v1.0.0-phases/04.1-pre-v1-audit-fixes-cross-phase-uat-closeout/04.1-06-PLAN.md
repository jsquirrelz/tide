```yaml
name: Live E2E (Claude-backed)

# Phase 04.1 P2.4 — workflow_dispatch ONLY per locked user decision. No
# schedule: trigger committed at v1.0. Operators forking the repo can wire
# their own ANTHROPIC_API_KEY repo secret and add a schedule trigger when

permissions: {}

jobs:
  live-e2e:
    permissions:
      contents: read
    runs-on: ubuntu-latest
    timeout-minutes: 15
    steps:
      - uses: actions/checkout@<SHA-pinned per existing test-e2e.yml>

      - uses: actions/setup-go@<SHA-pinned per existing test-e2e.yml>
        with:
          go-version-file: go.mod

      - name: Install kind
        run: |
          curl -Lo ./kind https://kind.sigs.k8s.io/dl/v0.31.0/kind-linux-amd64
          chmod +x ./kind
          sudo mv ./kind /usr/local/bin/kind

      - name: Run live Claude E2E (cost-capped)
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: make test-e2e-live
```
