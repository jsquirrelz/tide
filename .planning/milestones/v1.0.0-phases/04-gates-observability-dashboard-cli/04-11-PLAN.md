```go
func (h *EventsHandler) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    flusher, ok := w.(http.Flusher)
    if !ok { http.Error(w, "streaming unsupported", 500); return }
    w.Header().Set("Content-Type", "text/event-stream")
    w.Header().Set("Cache-Control", "no-cache")
    w.Header().Set("Connection", "keep-alive")
    w.Header().Set("X-Accel-Buffering", "no")  // Pitfall 23

    projectName := chi.URLParam(r, "name")
    lastEventID, _ := strconv.ParseInt(r.Header.Get("Last-Event-ID"), 10, 64)
    sub := h.Hub.Subscribe(projectName, lastEventID)
    defer h.Hub.Unsubscribe(sub)

    ticker := time.NewTicker(15 * time.Second)
    defer ticker.Stop()

    for {
        select {
        case <-r.Context().Done():
            return
        case <-ticker.C:
            fmt.Fprint(w, ": heartbeat\n\n")
            flusher.Flush()
        case ev := <-sub.Events():
            fmt.Fprintf(w, "id: %d\nevent: %s\ndata: %s\n\n", ev.ID, ev.Type, ev.JSON)
            flusher.Flush()
        }
    }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser SSE → backend handler | Long-running connection; client may walk away (browser tab close). Backend MUST detect via r.Context().Done() and clean up — Pitfall 22 is the failure mode. |
| Backend pods/log → apiserver | Backend's dashboard SA needs `get pods/log`; cluster RBAC governs this. Misconfigured RBAC → 403 → handler must surface friendly SSE error event. |
| Operator's ingress → backend SSE | nginx-ingress (and most proxies) buffer responses by default; Pitfall 23 mitigation is the X-Accel-Buffering header + heartbeat. |

| T-04-D4 | DoS (pods/log leak — Pitfall 22) | cmd/dashboard/api/logs_sse.go | mitigate | defer stream.Close + 5-min idle timeout + ctx.Done watcher + TestPodLogIdleTimeout with -race + goleak. |
