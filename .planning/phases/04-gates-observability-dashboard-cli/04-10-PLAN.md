<!-- controller-runtime manager.Runnable composition -->

```go
mgr, err := manager.New(restCfg, manager.Options{
    Scheme:                 scheme,
    Metrics:                metricsserver.Options{BindAddress: ":8081"},
    HealthProbeBindAddress: ":8081",
    LeaderElection:         false,  // dashboard is stateless read replica
})

// Register the HTTP server as a Runnable
mgr.Add(manager.RunnableFunc(func(ctx context.Context) error {
    srv := &http.Server{Addr: ":8080", Handler: r}
    go func() { <-ctx.Done(); srv.Shutdown(context.Background()) }()
    return srv.ListenAndServe()
}))

// /healthz gating on informer cache sync:
mgr.AddHealthzCheck("informer-synced", func(_ *http.Request) error {
    if !mgr.GetCache().WaitForCacheSync(context.Background()) {
        return errors.New("informer cache not yet synced")
    }
    return nil
})
```
