```ts
// dashboard/web/src/lib/sse.ts
export function useSSEStream(url: string) {
  const [events, setEvents] = useState<Event[]>([]);
  const [state, setState] = useState<"connecting" | "connected" | "reconnecting" | "offline">("connecting");
  const lastEventId = useRef(0);

  useEffect(() => {
    const es = new EventSource(url);
    es.onopen = () => setState("connected");
    es.onmessage = (e) => {
      lastEventId.current = parseInt(e.lastEventId);
      setEvents(prev => [...prev, JSON.parse(e.data)]);
    };
    es.onerror = () => {
      es.close();
      setState("reconnecting");
      // Reconnect with exponential backoff handled by EventSource native — browser default 3s
    };
    return () => es.close();  // Pitfall 22 client-side cleanup
  }, [url]);

  return { events, state, lastEventId };
}
```
