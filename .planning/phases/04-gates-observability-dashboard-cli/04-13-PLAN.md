type TaskNodeData = { status: string; attempt: number; waveIndex: number };
type TaskNode = Node<TaskNodeData, "task">;

function TaskNodeComponent({ data, selected }: NodeProps<TaskNode>) {
  return <div className="...">{data.status}</div>;
}

const nodeTypes = { task: TaskNodeComponent, /* ... */ };

function ExecutionDAGView({ planName }: { planName: string }) {
  const [nodes, setNodes] = useNodesState(initialNodes);
  const [edges, setEdges] = useEdgesState(initialEdges);
  const ready = useNodesInitialized();

  useEffect(() => {
    if (!ready) return;
    const positioned = applyDagreLayout(nodes, edges, "LR");
    setNodes(positioned);
  }, [ready, nodes, edges]);

  return <ReactFlow nodes={nodes} edges={edges} nodeTypes={nodeTypes} fitView />;
}
