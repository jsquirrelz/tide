- Tool vars (`$(KIND)`, `$(CONTAINER_TOOL)`, `$(KUBECTL)`, `$(KUSTOMIZE)`, `$(ENVTEST)`) live in the Dependencies block (lines 229-237).

**If adding helm as a tracked tool:** add `HELM ?= helm` near line 231 (alongside `KUBECTL ?= kubectl` and `KIND ?= kind` — system deps, not project-local binaries). Reference as `$(HELM)` and quote it as `"$(HELM)"` in recipes (matches the `"$(KUSTOMIZE)"` quoting pattern at line 430).

// applyYAML applies a YAML string to the kind cluster via kubectl stdin.
func applyYAML(yaml string) error {
    cmd := exec.CommandContext(ctx, "kubectl", "--kubeconfig", kubeconfigPath,
        "apply", "-f", "-", "--timeout=30s")
    cmd.Stdin = strings.NewReader(yaml)
    out, err := cmd.CombinedOutput()

        return fmt.Errorf("kubectl apply failed: %w\n%s", err, out)

    return nil
}
