<additional_context>
**Defense in depth:** the webhook denylist (`/v1/admin`, `/v1/billing`) is the PRIMARY gate. The credproxy is the RUNTIME ENFORCER. Both must reject overly-broad routes — webhook prevents admission of bad config; credproxy ensures the routes that were admitted are still subject to the hardcoded baseline.

**Hardcoded baseline NEVER removed:** operators can ADD routes via ExtraAllowedRoutes; they cannot REMOVE `/v1/messages` or `/v1/messages/count_tokens` (defense against config-attacks that try to surprise the orchestrator by dropping baseline routes).

// marshalAllowedRoutes serializes the flattened AllowedRoutes from all
// Project.Spec.Providers entries into a JSON array for the TIDE_ALLOWED_ROUTES
// env var consumed by the credproxy sidecar (Phase 04.1 P4.2).
//
// Returns "[]" on empty input — credproxy unmarshals this into an empty slice
// and only the hardcoded baseline allowlist applies.
func marshalAllowedRoutes(providers []tidev1alpha1.ProviderConfig) string {
    var routes []routeJSON
    for _, p := range providers {
        for _, r := range p.AllowedRoutes {
            routes = append(routes, routeJSON{Method: r.Method, PathPrefix: r.PathPrefix})
        }
    }
    b, err := json.Marshal(routes)
    if err != nil {
        return "[]"  // safe default — credproxy treats as empty
    }
    return string(b)
}

type routeJSON struct {
    Method     string `json:"method"`
    PathPrefix string `json:"pathPrefix"`
}
```

import (

    "sigs.k8s.io/controller-runtime/pkg/webhook/admission"

)

var projectlog = logf.Log.WithName("project-webhook")

// SetupProjectWebhookWithManager registers the Project validating webhook.
// Phase 04.1 P4.2: rejects PathPrefix values matching admin/billing surfaces
// regardless of operator configuration (defense in depth alongside the
// runtime credproxy enforcer).
func SetupProjectWebhookWithManager(mgr ctrl.Manager) error {
    return ctrl.NewWebhookManagedBy(mgr).
        For(&tideprojectv1alpha1.Project{}).
        WithValidator(&ProjectCustomValidator{}).
        Complete()
}

// +kubebuilder:webhook:path=/validate-tideproject-k8s-v1alpha1-project,mutating=false,failurePolicy=fail,sideEffects=None,groups=tideproject.k8s,resources=projects,verbs=create;update,versions=v1alpha1,name=vproject-v1alpha1.kb.io,admissionReviewVersions=v1

type ProjectCustomValidator struct{}

func (v *ProjectCustomValidator) ValidateCreate(ctx context.Context, obj admission.Object) (admission.Warnings, error) {
    project, ok := obj.(*tideprojectv1alpha1.Project)
    if !ok {
        return nil, fmt.Errorf("expected Project; got %T", obj)
    }
    return v.validate(project)
}

func (v *ProjectCustomValidator) ValidateUpdate(ctx context.Context, oldObj, newObj admission.Object) (admission.Warnings, error) {
    project, ok := newObj.(*tideprojectv1alpha1.Project)
    if !ok {
        return nil, fmt.Errorf("expected Project; got %T", newObj)
    }
    return v.validate(project)
}

func (v *ProjectCustomValidator) ValidateDelete(_ context.Context, _ admission.Object) (admission.Warnings, error) {
    return nil, nil
}

// validate enforces the credproxy-allowlist denylist (Phase 04.1 P4.2):
// PathPrefix matching admin/billing surfaces is rejected even if the operator
// tries to add them. The hardcoded denylist is intentionally narrow — only
// admin and billing paths. New denied surfaces should be added here, not via
// runtime config.
func (v *ProjectCustomValidator) validate(project *tideprojectv1alpha1.Project) (admission.Warnings, error) {
    denied := []string{"/v1/admin", "/v1/billing"}

    for i, prov := range project.Spec.Providers {
        for j, route := range prov.AllowedRoutes {
            for _, badPrefix := range denied {
                if route.PathPrefix == badPrefix || strings.HasPrefix(route.PathPrefix, badPrefix+"/") {
                    return nil, fmt.Errorf(
                        "spec.providers[%d].allowedRoutes[%d]: PathPrefix %q is on the denylist (Phase 04.1 P4.2 defense-in-depth)",
                        i, j, route.PathPrefix,
                    )
                }
            }
        }
    }
    return nil, nil
}
```
