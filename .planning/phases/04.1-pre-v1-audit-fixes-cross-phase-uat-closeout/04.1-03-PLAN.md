Both call sites (token mint in `task_controller.go:379-385`, activeDeadline derivation in `jobspec.go:108-113`) consume the helper, so any future drift between them is structurally impossible.

The bug: when `task.Spec.Caps` is nil or `WallClockSeconds <= 0`:
- Token mint (`task_controller.go:380-385`) applies the 300s floor and signs a token valid for 360s (300 + 60s grace)
- `BuildJobSpec` (`jobspec.go:108-113`) reads caps directly with no floor — `wallClockSeconds = 0`, `activeDeadline = 0 + 60 = 60s`

Result: the Pod is killed at t=60s while the token is still valid until t=360s. P1.3 fixes this by routing both derivations through one helper.

Per RESEARCH.md §P1.3 §"Confidence: HIGH — 30-line change, deterministic." The helper lands in `internal/dispatch/podjob/` (NOT `internal/controller/`) per the Architectural Responsibility Map — caps defaulting is a dispatch-tier concern even though the controller is one of the consumers.

Purpose: structural elimination of the token-validity-vs-deadline drift class.
Output: new `caps.go` + `caps_test.go` in podjob; updated `jobspec.go` and `task_controller.go` call sites; old constant deleted.

```go
package podjob

import (
    tidev1alpha1 "github.com/jsquirrelz/tide/api/v1alpha1"
)

// DefaultWallClockFloorSeconds is the minimum wall-clock budget applied when
// caps is nil or caps.WallClockSeconds <= 0. Sized to outlive image pull +
// scheduler delay + init container startup on a cold cluster (Phase 2 WR-01;
// hoisted from internal/controller/task_controller.go by Phase 04.1 P1.3).
//
// MUST match the value used by both the controller's token mint and the
// Job's activeDeadlineSeconds derivation, OR the token expires before the
// Pod can use it (token validity < deadline) or the Pod is killed mid-call
// (deadline < token validity). Phase 04.1 P1.3 enforces this structurally
// by routing both consumers through DefaultCaps below.
const DefaultWallClockFloorSeconds int32 = 300

// DefaultCaps returns a *Caps with the wall-clock floor applied. If caps is
// non-nil and WallClockSeconds > 0, returns caps unchanged (no allocation).
// Otherwise returns a NEW *Caps with WallClockSeconds = DefaultWallClockFloorSeconds
// and any non-zero fields from the input preserved (Iterations, InputTokens,
// OutputTokens — operator-set caps on other dimensions are NOT clobbered).
//
// Used by:
//   - internal/controller/task_controller.go (token mint — credproxy.Sign validity)
//   - internal/dispatch/podjob/jobspec.go    (activeDeadlineSeconds derivation)
//
// A nil-caps unit test (caps_test.go) asserts that both consumers' derived
// deadlines match within DefaultWallClockGraceSeconds, which the structural
// routing makes a tautology — the test fails only if a future maintainer
// routes one consumer around this helper.
func DefaultCaps(caps *tidev1alpha1.Caps) *tidev1alpha1.Caps {
    if caps != nil && caps.WallClockSeconds > 0 {
        return caps
    }
    out := tidev1alpha1.Caps{
        WallClockSeconds: DefaultWallClockFloorSeconds,
    }
    if caps != nil {
        if caps.Iterations > 0 {
            out.Iterations = caps.Iterations
        }
        if caps.InputTokens > 0 {
            out.InputTokens = caps.InputTokens
        }
        if caps.OutputTokens > 0 {
            out.OutputTokens = caps.OutputTokens
        }
    }
    return &out
}
```

```go
package podjob

import (
    "testing"

    tidev1alpha1 "github.com/jsquirrelz/tide/api/v1alpha1"
)

// TestDefaultCaps asserts the wall-clock floor application across nil, zero,
// and non-zero caps inputs. Covers Phase 04.1 P1.3 success criterion: the
// 300s floor is applied identically wherever DefaultCaps is called.
func TestDefaultCaps(t *testing.T) {
    cases := []struct {
        name             string
        in               *tidev1alpha1.Caps
        wantWallClock    int32
        wantIterations   int32
        wantInputTokens  int64
        wantOutputTokens int64
    }{
        {
            name:          "nil caps → 300s floor",
            in:            nil,
            wantWallClock: DefaultWallClockFloorSeconds,
        },
        {
            name:          "zero WallClockSeconds → 300s floor",
            in:            &tidev1alpha1.Caps{WallClockSeconds: 0},
            wantWallClock: DefaultWallClockFloorSeconds,
        },
        {
            name:          "negative WallClockSeconds → 300s floor",
            in:            &tidev1alpha1.Caps{WallClockSeconds: -1},
            wantWallClock: DefaultWallClockFloorSeconds,
        },
        {
            name:          "60s WallClockSeconds → 60s (under floor but operator-set is honored)",
            in:            &tidev1alpha1.Caps{WallClockSeconds: 60},
            wantWallClock: 60,
        },
        {
            name:          "600s WallClockSeconds → 600s",
            in:            &tidev1alpha1.Caps{WallClockSeconds: 600},
            wantWallClock: 600,
        },
        {
            name:           "zero WallClockSeconds + non-zero Iterations → 300s floor, Iterations preserved",
            in:             &tidev1alpha1.Caps{WallClockSeconds: 0, Iterations: 50},
            wantWallClock:  DefaultWallClockFloorSeconds,
            wantIterations: 50,
        },
        {
            name:             "zero WallClockSeconds + non-zero Token caps → 300s floor, tokens preserved",
            in:               &tidev1alpha1.Caps{WallClockSeconds: 0, InputTokens: 100000, OutputTokens: 50000},
            wantWallClock:    DefaultWallClockFloorSeconds,
            wantInputTokens:  100000,
            wantOutputTokens: 50000,
        },
    }

    for _, tc := range cases {
        t.Run(tc.name, func(t *testing.T) {
            got := DefaultCaps(tc.in)
            if got == nil {
                t.Fatalf("DefaultCaps returned nil; want non-nil")
            }
            if got.WallClockSeconds != tc.wantWallClock {
                t.Errorf("WallClockSeconds = %d; want %d", got.WallClockSeconds, tc.wantWallClock)
            }
            if got.Iterations != tc.wantIterations {
                t.Errorf("Iterations = %d; want %d", got.Iterations, tc.wantIterations)
            }
            if got.InputTokens != tc.wantInputTokens {
                t.Errorf("InputTokens = %d; want %d", got.InputTokens, tc.wantInputTokens)
            }
            if got.OutputTokens != tc.wantOutputTokens {
                t.Errorf("OutputTokens = %d; want %d", got.OutputTokens, tc.wantOutputTokens)
            }
        })
    }
}

// TestDefaultCaps_NilCapsDeadlineMatch asserts that for nil-caps input, the
// token-validity derivation and the Job activeDeadline derivation arrive at
// the same wall-clock value. This is a tautology by construction (both
// consumers route through DefaultCaps) — and that's the point. The test
// fails only if a future maintainer routes one consumer around the helper.
//
// Phase 04.1 P1.3 success criterion: "the two derived deadlines match
// within a grace window" (the grace is DefaultWallClockGraceSeconds, applied
// equally to both consumers downstream of DefaultCaps).
func TestDefaultCaps_NilCapsDeadlineMatch(t *testing.T) {
    capsForToken := DefaultCaps(nil)
    capsForJob := DefaultCaps(nil)

    tokenValidity := capsForToken.WallClockSeconds + DefaultWallClockGraceSeconds
    activeDeadline := capsForJob.WallClockSeconds + DefaultWallClockGraceSeconds

    if tokenValidity != activeDeadline {
        t.Errorf("token validity (%ds) != active deadline (%ds); Phase 04.1 P1.3 drift",
            tokenValidity, activeDeadline)
    }
    // Phase 04.1 P1.3: the floor is 300s and the grace is 60s, so both should be 360s.
    if tokenValidity != DefaultWallClockFloorSeconds+DefaultWallClockGraceSeconds {
        t.Errorf("expected nil-caps deadline = %ds (floor + grace); got %ds",
            DefaultWallClockFloorSeconds+DefaultWallClockGraceSeconds, tokenValidity)
    }
}
```
